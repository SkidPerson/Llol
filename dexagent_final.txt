-- Tabela para guardar os botoes criados
local playerButtons = {}

-- Funcao para criar um botao de player
local function createPlayerButton(playerName, isLocalPlayer, index)
    local btn = Instance.new("TextButton")
    btn.Name = "PlayerBtn_" .. playerName
    btn.Parent = LMG2L["ScrollingFrame_3"]
    btn.BorderSizePixel = 0
    btn.TextXAlignment = Enum.TextXAlignment.Left
    btn.TextSize = 16
    btn.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    btn.FontFace = Font.new([[rbxasset://fonts/families/SourceSansPro.json]], Enum.FontWeight.Regular, Enum.FontStyle.Normal)
    btn.BackgroundTransparency = 1
    btn.Size = UDim2.new(0, 172, 0, 18)
    btn.Position = UDim2.new(0, 78, 0, 24 + (index * 18))
    btn.Visible = false  -- Comeca escondido
    
    -- Se for voce, destaca em amarelo/dourado
    if isLocalPlayer then
        btn.Text = playerName
        btn.TextColor3 = Color3.fromRGB(211, 211, 211) -- Dourado
    else
        btn.Text = playerName
        btn.TextColor3 = Color3.fromRGB(211, 211, 211) -- Cinza claro
    end
    
    -- Aumenta o tamanho do ScrollingFrame se necessario
    local totalHeight = 24 + ((index + 1) * 18)
    if totalHeight > LMG2L["ScrollingFrame_3"].CanvasSize.Y.Offset then
        LMG2L["ScrollingFrame_3"].CanvasSize = UDim2.new(0, 0, 0, totalHeight + 10)
    end
    
    table.insert(playerButtons, btn)
    return btn
end

-- Funcao para atualizar a lista de players
local function updatePlayerList()
    -- Remove botoes antigos
    for _, btn in ipairs(playerButtons) do
        btn:Destroy()
    end
    playerButtons = {}
    
    local index = 0
    
    -- Primeiro: VOCE (LocalPlayer) em dourado
    createPlayerButton(LocalPlayer.Name, true, index)
    index = index + 1
    
    -- Depois: todos os outros players do servidor
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            createPlayerButton(player.Name, false, index)
            index = index + 1
        end
    end
end

-- Atualiza quando entra/sai player
Players.PlayerAdded:Connect(updatePlayerList)
Players.PlayerRemoving:Connect(updatePlayerList)

-- Atualiza lista inicial
updatePlayerList()

-- DRAG (MOVE WINDOW) - TOPBAR

local dragging = false
local dragStartPos = nil
local windowStartPos = nil

EditorTopbar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStartPos = input.Position
        windowStartPos = EditorWindow.Position
    end
end)

EditorTopbar.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragging then
            local delta = input.Position - dragStartPos
            EditorWindow.Position = UDim2.new(
                windowStartPos.X.Scale,
                windowStartPos.X.Offset + delta.X,
                windowStartPos.Y.Scale,
                windowStartPos.Y.Offset + delta.Y
            )
        end
    end
end)

            local newH = math.max(180, resizeStartSize.Y.Offset + delta.Y)
            EditorWindow.Size = UDim2.new(0, newW, 0, newH)
        end
    end
end)

-- RUN / CLEAR

EditorRunBtn.MouseButton1Click:Connect(function()
    local code = EditorTextBox.Text
    if code == "" then
        EditorOutput.Text = "> Nada para executar."
        EditorOutput.TextColor3 = Color3.fromRGB(200, 180, 60)
        return
    end
    local ok, err = pcall(function()
        local fn, loadErr = loadstring(code)
        if fn then
            fn()
            EditorOutput.Text = "> Executado com sucesso."
            EditorOutput.TextColor3 = Color3.fromRGB(100, 200, 100)
        else
            EditorOutput.Text = "> Erro: " .. tostring(loadErr)
            EditorOutput.TextColor3 = Color3.fromRGB(220, 80, 80)
        end
    end)
    if not ok then
        EditorOutput.Text = "> Erro ao executar."
        EditorOutput.TextColor3 = Color3.fromRGB(220, 80, 80)
    end
end)

EditorClearBtn.MouseButton1Click:Connect(function()
    EditorTextBox.Text = ""
    EditorOutput.Text = "> Limpo."
    EditorOutput.TextColor3 = Color3.fromRGB(100, 200, 100)
end)
